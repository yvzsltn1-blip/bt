<script>
const SKILLS=[{key:'strength',label:'Güç',icon:'⚔️'},{key:'defense',label:'Savunma',icon:'🛡️'},{key:'dexterity',label:'Beceri',icon:'🎯'},{key:'endurance',label:'Dayanıklılık',icon:'🩸'},{key:'charisma',label:'Karizma',icon:'🦇'}];
let mode='target', selected='strength', discount=false, strategy='equal', selectedMulti=new Set(SKILLS.map(s=>s.key));
const $=id=>document.getElementById(id); const fmt=n=>new Intl.NumberFormat('tr-TR').format(Math.max(0,Math.floor(n||0))); const clean=v=>{const n=parseInt(String(v||'').replace(/\D/g,''),10);return Number.isFinite(n)?n:0};
function cost(skill){skill=Math.max(5,Number(skill)||5);let v=Math.pow(skill-4,2.4);return Math.floor(discount?v*.4:v)}
function total(from,to){from=Math.max(5,clean(from));to=Math.max(from,clean(to));let s=0;for(let i=from;i<to;i++)s+=cost(i);return s}
function buy(start,gold){let skill=Math.max(5,clean(start)),rem=Math.max(0,clean(gold)),spent=0,rows=[];while(true){let c=cost(skill);if(c>rem)break;rem-=c;spent+=c;rows.push({from:skill,to:skill+1,cost:c});skill++;if(rows.length>50000)break}return{final:skill,gain:rows.length,spent,rem,rows}}
function card(icon,title,value,sub){return `<div class="card metal"><div class="head"><div class="icon">${icon}</div>${title}</div><div class="big">${value}</div><div class="tiny">${sub||''}</div></div>`}
function renderSkillGrid(id,multi=false){$(id).innerHTML=SKILLS.map(s=>{let active=multi?selectedMulti.has(s.key):selected===s.key;return `<button class="skill ${active?'active':''}" data-key="${s.key}"><div class="emoji">${s.icon}</div><div class="name">${s.label}</div></button>`}).join('');$(id).querySelectorAll('.skill').forEach(b=>b.onclick=()=>{if(multi){selectedMulti.has(b.dataset.key)?selectedMulti.delete(b.dataset.key):selectedMulti.add(b.dataset.key)}else selected=b.dataset.key;render()})}
function renderSingle(){let sk=SKILLS.find(s=>s.key===selected),cur=clean($('current').value),tar=clean($('target').value),gold=clean($('gold').value);$('currentLabel').textContent='Mevcut '+sk.label;$('targetLabel').textContent='Hedef '+sk.label; if(mode==='target'){let c=total(cur,tar),normal=discount?(()=>{let old=discount;discount=false;let n=total(cur,tar);discount=old;return n})():c;$('results').innerHTML=card('🪙',`${cur} → ${tar} toplam maliyet`,fmt(c)+' altın',discount?'Normal fiyat: '+fmt(normal)+' altın':'İndirimsiz normal eğitim maliyeti')+card('📈','Alınacak skill puanı','+'+fmt(Math.max(0,tar-cur)),sk.label+' için hesaplandı')+card('%','Etkinlik kazancı',fmt(normal-c)+' altın',discount?'İndirim sayesinde cebinde kalır':'İndirim kapalı olduğu için 0')}else{let r=buy(cur,gold);let list=r.rows.slice(0,6).map(x=>`<div class="list-row"><span>${x.from} → ${x.to}</span><b>${fmt(x.cost)} altın</b></div>`).join('')||'<div class="tiny">Bu altın bir sonraki puana yetmiyor.</div>';$('results').innerHTML=card('📈','Ulaşacağın skill',fmt(r.final)+' '+sk.label,'Toplam +'+fmt(r.gain)+' skill puanı')+card('🪙','Harcanan / kalan',fmt(r.spent)+' altın','Kalan: '+fmt(r.rem)+' altın')+`<div class="card metal"><div class="head"><div class="icon">🧮</div>İlk basımlar</div><div class="list">${list}</div>${r.rows.length>6?`<div class="tiny">+ ${fmt(r.rows.length-6)} satır daha</div>`:''}</div>`}}
function multiVals(){let o={};SKILLS.forEach(s=>o[s.key]=clean($('m_'+s.key).value));return o}
function equalDist(vals,gold){let keys=[...selectedMulti],share=keys.length?Math.floor(gold/keys.length):0,left=keys.length?gold-share*keys.length:gold;let res=SKILLS.map(s=>{let r=keys.includes(s.key)?buy(vals[s.key],share):buy(vals[s.key],0);return{...s,start:vals[s.key],sel:keys.includes(s.key),...r,share:keys.includes(s.key)?share:0}});return{res,spent:res.reduce((a,b)=>a+b.spent,0),gain:res.reduce((a,b)=>a+b.gain,0),rem:res.reduce((a,b)=>a+b.rem,0)+left,share}}
function optDist(vals,gold){let rem=gold,res=SKILLS.map(s=>({...s,start:vals[s.key],final:Math.max(5,vals[s.key]),spent:0,gain:0,rem:0,sel:selectedMulti.has(s.key)}));let safe=0;while(true){let cand=res.filter(x=>x.sel).map(x=>({...x,next:cost(x.final)})).sort((a,b)=>a.next-b.next)[0];if(!cand||cand.next>rem)break;let i=res.findIndex(x=>x.key===cand.key);res[i].spent+=cand.next;res[i].final++;res[i].gain++;rem-=cand.next;if(++safe>50000)break}res.forEach(x=>x.rem=0);return{res,spent:res.reduce((a,b)=>a+b.spent,0),gain:res.reduce((a,b)=>a+b.gain,0),rem,share:0}}
function renderMulti(){let vals=multiVals(),gold=clean($('m_gold').value),r=strategy==='equal'?equalDist(vals,gold):optDist(vals,gold);let rows=r.res.map(x=>`<div class="list-row" style="opacity:${x.sel?1:.58}"><span><b style="color:#fff">${x.icon} ${x.label}</b><br><small style="color:#94a3b8">${fmt(x.start)} → ${fmt(x.final)}</small></span><span style="text-align:right"><b>+${fmt(x.gain)}</b><br><small style="color:#94a3b8">${fmt(x.spent)} altın</small></span></div>`).join('');$('results').innerHTML=card(strategy==='equal'?'⚖️':'✨',strategy==='equal'?'Eşit dağıtım özeti':'Optimum dağıtım özeti','+'+fmt(r.gain)+' skill','Harcanan: '+fmt(r.spent)+' altın • Kalan: '+fmt(r.rem)+' altın')+card('🎛️','Seçili skill sayısı',selectedMulti.size+' skill',strategy==='equal'?'Skill başına pay: '+fmt(r.share)+' altın':'Altın en ucuz sıradaki skill basımına gider')+`<div class="card metal"><div class="head"><div class="icon">🎯</div>Skill bazlı sonuçlar</div><div class="list">${rows}</div></div>`}
function render(){renderSkillGrid('singleSkills');renderSkillGrid('multiSkills',true);$('discount').classList.toggle('on',discount);$('tabTarget').classList.toggle('active',mode==='target');$('tabBudget').classList.toggle('active',mode==='budget');$('tabMulti').classList.toggle('active',mode==='multi');$('singleBox').classList.toggle('hidden',mode==='multi');$('multiBox').classList.toggle('hidden',mode!=='multi');$('targetField').classList.toggle('hidden',mode!=='target');$('goldField').classList.toggle('hidden',mode!=='budget');$('equalBtn').classList.toggle('active',strategy==='equal');$('optBtn').classList.toggle('active',strategy==='opt');mode==='multi'?renderMulti():renderSingle()}
['current','target','gold','m_strength','m_defense','m_dexterity','m_endurance','m_charisma','m_gold'].forEach(id=>$(id).oninput=e=>{e.target.value=e.target.value.replace(/\D/g,'');render()});
$('tabTarget').onclick=()=>{mode='target';render()};$('tabBudget').onclick=()=>{mode='budget';render()};$('tabMulti').onclick=()=>{mode='multi';render()};$('discount').onclick=()=>{discount=!discount;render()};$('equalBtn').onclick=()=>{strategy='equal';render()};$('optBtn').onclick=()=>{strategy='opt';render()};$('presetAll').onclick=()=>{selectedMulti=new Set(SKILLS.map(s=>s.key));render()};$('presetStrDex').onclick=()=>{selectedMulti=new Set(['strength','dexterity']);render()};$('presetFight').onclick=()=>{selectedMulti=new Set(['strength','defense','dexterity']);render()};$('presetTank').onclick=()=>{selectedMulti=new Set(['defense','endurance']);render()};$('resetSingle').onclick=()=>{$('current').value=500;$('target').value=503;$('gold').value=500000000;selected='strength';render()};$('copyExample').onclick=()=>{$('current').value=500;$('target').value=503;mode='target';render()};$('resetMulti').onclick=()=>{SKILLS.forEach(s=>$('m_'+s.key).value=500);$('m_gold').value=500000000;selectedMulti=new Set(SKILLS.map(s=>s.key));strategy='equal';render()};
render();
</script>
